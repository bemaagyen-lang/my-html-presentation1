      import pandas as pd
      import seaborn as sns
      import matplotlib.pyplot as plt
      # --------------------------------------------------------
      # 2. Calculate completeness and missing %
      # --------------------------------------------------------
      total_records = len(data_dist1)
      missing = data_dist1['runoff_mean'].isna().sum()
      complete = total_records - missing
      missing_pct = (missing / total_records) * 100
      complete_pct = (complete / total_records) * 100
      summary = data_dist1.groupby('station_name')['runoff_mean'].agg(
          total_records='size',
          missing_values=lambda x: x.isna().sum(),
          complete_values=lambda x: x.notna().sum()
      )
      # Top 3 stations with most complete data
      top3_complete = summary.sort_values("complete_values", ascending=False).head(2)
      top3_list = ", ".join(top3_complete.index)
      # Station with the most missing data
      most_missing = summary.sort_values("missing_values", ascending=False).head(2)
      top3_least = ", ".join(most_missing.index)
      # --------------------------------------------------------
      # 3. Donut Chart + Text Box Side by Side
      # --------------------------------------------------------
      fig, axs = plt.subplots(1, 2, figsize=(14, 7))  # CREATE TWO SUBPLOTS
      # --------------------------
      # LEFT PANEL: DONUT PLOT
      # --------------------------
      labels = ['Missing Data', 'Complete Data']
      sizes = [missing_pct, complete_pct]
      wedges, texts, autotexts = axs[0].pie(
          sizes,
          labels=labels,
          autopct='%1.1f%%',
          startangle=90,
          pctdistance=0.85,
      )
      # Donut hole
      centre_circle = plt.Circle((0, 0), 0.70, fc='white')
      axs[0].add_artist(centre_circle)
      axs[0].axis('equal')
      axs[0].set_title("Nationwide  Hydro-Data Records", fontsize=16)
      axs[0].legend(loc='lower center')
      # --------------------------
      # RIGHT PANEL: TEXT BOX
      # --------------------------
      def classify_region(geo_y):
          if geo_y >= 8.5:
              return "North"
          else:
              return "South"
      data_dist1.loc[:,"region"] = data_dist1["geo_y"].apply(classify_region)
      # Verify classification
      #print(data_dist1[["station_name", "geo_y", "region"]].drop_duplicates())
      dd = data_dist1[["station_name", "geo_y", "region"]].drop_duplicates()
      count= dd["region"].count()
         
      summary_text = (
          f"HYDROLOGICAL DATA SUMMARY\n\n"
         
          f" Total avail. records:      {total_records:,}\n"
          f" Number of stations:           {count:,}\n"
          f" stations with  most complete data: {top3_list}\n"
          f" Station with most missing data:    {top3_least}\n"
      )
      axs[1].text(
          0.02, 0.8, summary_text,
          fontsize=13,
          va='center',
          ha='left',
          wrap=True,
          bbox=dict(facecolor='whitesmoke', edgecolor='black', boxstyle='round,pad=1')
      )
      axs[1].axis('off')
      plt.tight_layout()
      plt.show()
